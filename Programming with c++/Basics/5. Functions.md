## 1. Function Declaration and Definition

### Basic Syntax

```cpp
// Declaration (prototype)
return_type function_name(parameter_list);

// Definition
return_type function_name(parameter_list) 
{
    // function body
    return value; // if not void
}
```

### Examples

```cpp
#include <iostream>
using namespace std;

// Function declaration
int add(int a, int b);
void greet(string name);
double calculateArea(double radius);

// Function definitions
int add(int a, int b) 
{
    return a + b;
}

void greet(string name) 
{
    cout << "Hello, " << name << "!" << endl;
}

double calculateArea(double radius) 
{
    return 3.14159 * radius * radius;
}

int main() 
{
    int result = add(5, 3);          // 8
    greet("Alice");                  // Hello, Alice!
    double area = calculateArea(5.0); // 78.5398
    
    cout << "Sum: " << result << endl;
    cout << "Area: " << area << endl;
    
    return 0;
}
```

## 2. Parameters and Return Values

### Different Parameter Types

```cpp
// No parameters
void sayHello() 
{
    cout << "Hello World!" << endl;
}

// Single parameter
int square(int num) 
{
    return num * num;
}

// Multiple parameters
int multiply(int a, int b, int c) 
{
    return a * b * c;
}

// Different return types
string getName() 
{
    return "John Doe";
}

bool isEven(int num) 
{
    return num % 2 == 0;
}

// Void return (no return value)
void printMessage(string msg) 
{
    cout << "Message: " << msg << endl;
}
```

### Using Functions

```cpp
int main() 
{
    sayHello();                    // Hello World!
    
    int num = 5;
    cout << square(num) << endl;   // 25
    
    int product = multiply(2, 3, 4); // 24
    
    string name = getName();       // "John Doe"
    
    if (isEven(10)) 
    {
        cout << "10 is even" << endl;
    }
    
    printMessage("Learning C++");  // Message: Learning C++
    
    return 0;
}
```

## 3. Default Arguments

### Basic Default Arguments

```cpp
// Function with default parameters
void printInfo(string name, int age = 25, string city = "Unknown") 
{
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    cout << "City: " << city << endl;
    cout << "---" << endl;
}

int main() 
{
    printInfo("Alice");                    // Uses default age and city
    printInfo("Bob", 30);                  // Uses default city only
    printInfo("Charlie", 35, "New York");  // No defaults used
    
    return 0;
}
```

### Rules for Default Arguments

```cpp
// ✅ Correct: Default arguments at the end
void func1(int a, int b = 10, int c = 20);

// ❌ Wrong: Non-default after default
// void func2(int a = 5, int b, int c = 20);

// Example function
int power(int base, int exponent = 2) 
{
    int result = 1;
    for (int i = 0; i < exponent; i++) 
    {
        result *= base;
    }
    return result;
}

int main() 
{
    cout << power(3) << endl;      // 9 (3^2, uses default)
    cout << power(2, 4) << endl;   // 16 (2^4)
    
    return 0;
}
```

## 4. Function Overloading

### Same Name, Different Parameters

```cpp
#include <iostream>
using namespace std;

// Overloaded functions
int add(int a, int b) 
{
    return a + b;
}

double add(double a, double b) 
{
    return a + b;
}

int add(int a, int b, int c) 
{
    return a + b + c;
}

string add(string a, string b) 
{
    return a + b;
}


int main() {
    cout << add(5, 3) << endl;              // int version: 8
    cout << add(2.5, 3.7) << endl;          // double version: 6.2
    cout << add(1, 2, 3) << endl;           // three-parameter version: 6
    cout << add("Hello", " World") << endl; // string version: Hello World
    
    return 0;
}
```

### Overloading Rules

```cpp
// Valid overloading (different parameter types/count)
void print(int x);
void print(double x);
void print(string x);
void print(int x, int y);

// ❌ Invalid: Only return type differs
// int getValue();
// double getValue(); // Error!

// Example: Area calculation overloading
double area(double radius) 
{                    
    return 3.14159 * radius * radius; // Circle
}

double area(double length, double width) 
{      
    return length * width; // Rectangle
}

double area(double a, double b, double c) 
{     
    double s = (a + b + c) / 2; // Triangle (Heron's formula)
    return sqrt(s * (s - a) * (s - b) * (s - c));
}
```

## 5. Inline Functions

### Basic Inline Functions

```cpp
// Inline function - compiler replaces call with function body
inline int max(int a, int b) 
{
    return (a > b) ? a : b;
}

inline double square(double x) 
{
    return x * x;
}

int main() 
{
    int larger = max(10, 20);     // Compiler replaces with: (10 > 20) ? 10 : 20
    double sq = square(5.5);      // Compiler replaces with: 5.5 * 5.5
    
    cout << "Larger: " << larger << endl;
    cout << "Square: " << sq << endl;
    
    return 0;
}
```

### When to Use Inline

```cpp
// Good candidates for inline (small, frequently called)
inline int abs(int x) 
{
    return (x < 0) ? -x : x;
}

inline bool isPositive(int x) 
{
    return x > 0;
}

// Poor candidates for inline (large functions)
inline void complexFunction() 
{
    // Many lines of code...
    // Compiler may ignore inline request
}
```

## 6. Recursion

### Basic Recursion

```cpp
// Factorial: n! = n * (n-1)!
int factorial(int n) 
{
    // Base case
    if (n <= 1) 
    {
        return 1;
    }
    // Recursive case
    return n * factorial(n - 1);
}

// Fibonacci: fib(n) = fib(n-1) + fib(n-2)
int fibonacci(int n) 
{
    if (n <= 1) 
    {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

int main() 
{
    cout << "5! = " << factorial(5) << endl;           // 120
    cout << "Fibonacci(6) = " << fibonacci(6) << endl; // 8
    
    return 0;
}
```

### Recursion vs Iteration

```cpp
// Recursive sum
int recursiveSum(int n) 
{
    if (n <= 1) return n;
    return n + recursiveSum(n - 1);
}

// Iterative sum
int iterativeSum(int n) 
{
    int sum = 0;
    for (int i = 1; i <= n; i++) 
    {
        sum += i;
    }
    return sum;
}

// Both give same result, but iterative is usually more efficient
```

## 7. Variable Scope and Lifetime

### Local Variables

```cpp
void function1() 
{
    int localVar = 10; // Local to this function
    cout << localVar << endl;
} // localVar destroyed here

void function2() 
{
    // cout << localVar;  // Error! localVar not accessible
    int localVar = 20;   // Different variable with same name
    cout << localVar << endl;
}
```

### Global Variables

```cpp
#include <iostream>
using namespace std;

int globalVar = 100;  // Global variable

void showGlobal() 
{
    cout << "Global: " << globalVar << endl;  // Can access global
}

void modifyGlobal() 
{
    globalVar = 200;  // Can modify global
}

int main() 
{
    cout << "Initial: " << globalVar << endl;  // 100
    showGlobal();                              // 100
    modifyGlobal();
    showGlobal();                              // 200
    
    return 0;
}
```

### Static Variables

```cpp
void counter() 
{
    static int count = 0;  // Static variable - retains value between calls
    count++;
    cout << "Count: " << count << endl;
}

int main() 
{
    counter();  // Count: 1
    counter();  // Count: 2
    counter();  // Count: 3
    
    return 0;
}
```

### Scope Hierarchy

```cpp
int global = 1;           // Global scope

int main() 
{
    int local = 2;        // Function scope
    
    {
        int block = 3;    // Block scope
        cout << global << local << block << endl;  // Can access all
    }
    // cout << block;     // Error! block is out of scope
    
    cout << global << local << endl;  // OK
    
    return 0;
}
```

## 8. Passing by Value vs. Passing by Reference

### Pass by Value (Default)

```cpp
void passByValue(int x) 
{
    x = 100;  // Only changes local copy
    cout << "Inside function: " << x << endl;  // 100
}

int main() 
{
    int num = 50;
    cout << "Before: " << num << endl;    // 50
    passByValue(num);                     // 100
    cout << "After: " << num << endl;     // Still 50 (unchanged)
    
    return 0;
}
```

### Pass by Reference

```cpp
void passByReference(int &x) // & makes it a reference
{  
    x = 100;  // Changes original variable
    cout << "Inside function: " << x << endl;  // 100
}

int main() 
{
    int num = 50;
    cout << "Before: " << num << endl;     // 50
    passByReference(num);                  // 100
    cout << "After: " << num << endl;      // 100 (changed!)
    
    return 0;
}
```

### Practical Example: Swap Function

```cpp
// Won't work - pass by value
void swapByValue(int a, int b) 
{
    int temp = a;
    a = b;
    b = temp;
    // Only swaps local copies
}

// Works - pass by reference
void swapByReference(int &a, int &b) 
{
    int temp = a;
    a = b;
    b = temp;
    // Swaps original variables
}

int main() 
{
    int x = 10, y = 20;
    
    cout << "Before: x=" << x << ", y=" << y << endl;  // x=10, y=20
    
    swapByValue(x, y);
    cout << "After value swap: x=" << x << ", y=" << y << endl;  
    // x = 10, y = 20 (no change)
    
    swapByReference(x, y);
    cout << "After reference swap: x=" << x << ", y=" << y << endl;  
    // x = 20, y = 10 (swapped!)
    
    return 0;
}
```

### When to Use Each

```cpp
// Use pass by value when:
// - You don't want to modify the original
// - Working with small data types (int, char, bool)

int calculateSquare(int num) // Safe, won't change original
{  
    return num * num;
}

// Use pass by reference when:
// - You want to modify the original variable
// - Working with large objects (to avoid copying)

void updateScore(int &score, int points) 
{
    score += points;  // Modifies original score
}
```

## Quick Reference

### Function Template

```cpp
return_type functionName(parameters) 
{
    // function body
    return value;  // if not void
}
```

### Key Points

- **Declaration before use** (or define before main)
- **Default arguments** must be at the end
- **Overloading** based on parameters, not return type
- **Inline** for small, frequently called functions
- **Recursion** needs base case to avoid infinite calls
- **Global variables** accessible everywhere
- **Static variables** retain value between calls
- **Pass by reference (&)** to modify original values

### Common Patterns

```cpp
// Input validation
bool isValidAge(int age) 
{
    return age >= 0 && age <= 150;
}

// Mathematical operations
double power(double base, int exp) {

    double result = 1;
    for (int i = 0; i < exp; i++) 
    {
        result *= base;
    }
    return result;
}

// Array processing
void printArray(int arr[], int size) 
{
    for (int i = 0; i < size; i++) 
    {
        cout << arr[i] << " ";
    }
    cout << endl;
}
```
