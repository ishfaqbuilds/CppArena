### 1. Basic String Operations

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() 
{
    // String declaration and initialization
    string str1 = "Hello";
    string str2("World");
    string str3;  // Empty string
    
    // String concatenation
    string greeting = str1 + " " + str2;  // "Hello World"
    cout << greeting << endl;
    
    // String assignment
    str3 = "C++ Programming";
    
    // Access characters
    cout << "First char: " << str1[0] << endl;        // 'H'
    cout << "Last char: " << str2[str2.length()-1] << endl;  // 'd'
    
    // Modify characters
    str1[0] = 'h';  // "hello"
    cout << "Modified: " << str1 << endl;
    
    // String comparison
    if (str1 == "hello") 
    {
        cout << "Strings are equal" << endl;
    }
    
    return 0;
}
```

## 2. String Functions

### Essential String Methods

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string text = "C++ Programming";
    
    // Length/Size
    cout << "Length: " << text.length() << endl;    // 15
    cout << "Size: " << text.size() << endl;        // 15 (same as length)
    
    // Check if empty
    string empty = "";
    cout << "Is empty: " << empty.empty() << endl;  // 1 (true)
    
    // Substring
    string sub = text.substr(4, 11);  // From index 4, length 11
    cout << "Substring: " << sub << endl;  // "Programming"
    
    // Find
    size_t pos = text.find("Program");
    if (pos != string::npos) 
    {
        cout << "Found at position: " << pos << endl;  // 4
    }
    
    // Replace
    string replaced = text;
    replaced.replace(0, 3, "Java");  // Replace "C++" with "Java"
    cout << "Replaced: " << replaced << endl;  // "Java Programming"
    
    return 0;
}
```

### String Search and Manipulation

```cpp
int main() 
{
    string sentence = "The quick brown fox jumps over the lazy dog";
    
    // Find first occurrence
    size_t pos1 = sentence.find("the");
    cout << "First 'the' at: " << pos1 << endl;  // 31
    
    // Find from specific position
    size_t pos2 = sentence.find("the", 0);  // Start from beginning
    cout << "First 'the' from start: " << pos2 << endl;  // 31
    
    // Find last occurrence
    size_t pos3 = sentence.rfind("the");
    cout << "Last 'the' at: " << pos3 << endl;  // 31
    
    // Find any character from a set
    size_t vowel = sentence.find_first_of("aeiou");
    cout << "First vowel at: " << vowel << endl;
    
    // Count occurrences of a character
    int count = 0;
    for (char c : sentence) 
    {
        if (c == 'o') count++;
    }
    cout << "Letter 'o' appears " << count << " times" << endl;
    
    return 0;
}
```

## 3. C-style Strings (Character Arrays)

### Basic Character Arrays

```cpp
#include <iostream>
#include <cstring>  // For C-string functions
using namespace std;

int main() {
    // Character array declaration
    char str1[20] = "Hello";        // Null-terminated
    char str2[] = {'W', 'o', 'r', 'l', 'd', '\0'};  // Manual null terminator
    char str3[10];                  // Uninitialized
    
    // Display C-strings
    cout << "str1: " << str1 << endl;  // "Hello"
    cout << "str2: " << str2 << endl;  // "World"
    
    // Input C-string
    cout << "Enter a word: ";
    cin >> str3;  // Reads until whitespace
    cout << "You entered: " << str3 << endl;
    
    // Input with spaces
    char sentence[100];
    cout << "Enter a sentence: ";
    cin.ignore();
    cin.getline(sentence, 100);  // Read entire line
    cout << "Sentence: " << sentence << endl;
    
    return 0;
}
```

### Character Array Traversal

```cpp
int main() 
{
    char word[] = "Programming";
    
    // Method 1: Using strlen
    cout << "Using strlen: ";
    for (int i = 0; i < strlen(word); i++) 
    {
        cout << word[i];
    }
    cout << endl;
    
    // Method 2: Until null terminator
    cout << "Until null: ";
    for (int i = 0; word[i] != '\0'; i++) 
    {
        cout << word[i];
    }
    cout << endl;
    
    // Method 3: Pointer arithmetic
    cout << "Using pointers: ";
    for (char* ptr = word; *ptr != '\0'; ptr++) 
    {
        cout << *ptr;
    }
    cout << endl;
    
    return 0;
}
```

## 4. String Manipulation Functions

### Essential C-String Functions

```cpp
#include <iostream>
#include <cstring>
using namespace std;

int main() 
{
    char str1[50] = "Hello";
    char str2[50] = "World";
    char str3[50];
    
    // strlen - string length
    cout << "Length of str1: " << strlen(str1) << endl;  // 5
    
    // strcpy - string copy
    strcpy(str3, str1);  // Copy str1 to str3
    cout << "str3 after copy: " << str3 << endl;  // "Hello"
    
    // strcat - string concatenation
    strcat(str1, " ");   // str1 becomes "Hello "
    strcat(str1, str2);  // str1 becomes "Hello World"
    cout << "After concatenation: " << str1 << endl;
    
    // strcmp - string comparison
    int result = strcmp("Apple", "Banana");
    if (result < 0) 
    {
        cout << "Apple comes before Banana" << endl;
    } 
    else if (result > 0) 
    {
        cout << "Apple comes after Banana" << endl;
    } 
    else 
    {
        cout << "Strings are equal" << endl;
    }
    
    return 0;
}
```

### Advanced C-String Functions

```cpp
int main() 
{
    char source[] = "Programming in C++";
    char dest[50];
    char search[] = "gram";
    
    // strncpy - safe copy with limit
    strncpy(dest, source, 10);
    dest[10] = '\0';  // Ensure null termination
    cout << "Partial copy: " << dest << endl;  // "Programmin"
    
    // strncat - safe concatenation with limit
    char greeting[50] = "Hello ";
    strncat(greeting, "World", 3);  // Add only 3 characters
    cout << "Limited concat: " << greeting << endl;  // "Hello Wor"
    
    // strstr - find substring
    char* found = strstr(source, search);
    if (found != nullptr) 
    {
        cout << "Substring found at position: " << (found - source) << endl;
    }
    
    // strchr - find character
    char* charPos = strchr(source, 'g');
    if (charPos != nullptr) 
    {
        cout << "Character 'g' found at: " << (charPos - source) << endl;
    }
    
    return 0;
}
```

### String Utilities

```cpp
#include <cctype>  // For character functions

void toUpperCase(char str[]) 
{
    for (int i = 0; str[i] != '\0'; i++)
    {
        str[i] = toupper(str[i]);
    }
}

void toLowerCase(char str[]) 
{
    for (int i = 0; str[i] != '\0'; i++) 
    {
        str[i] = tolower(str[i]);
    }
}

bool isPalindrome(char str[]) 
{
    int len = strlen(str);
    for (int i = 0; i < len / 2; i++) 
    {
        if (str[i] != str[len - 1 - i]) 
        {
            return false;
        }
    }
    return true;
}

int main() 
{
    char text[] = "Hello World";
    
    // Convert to uppercase
    toUpperCase(text);
    cout << "Uppercase: " << text << endl;  // "HELLO WORLD"
    
    // Convert to lowercase
    toLowerCase(text);
    cout << "Lowercase: " << text << endl;  // "hello world"
    
    // Check palindrome
    char palindromeTest[] = "racecar";
    if (isPalindrome(palindromeTest)) 
    {
        cout << palindromeTest << " is a palindrome" << endl;
    }
    
    return 0;
}
```

## 5. Comparison: std::string vs C-style strings

### std::string Advantages

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    // Easy concatenation
    string str1 = "Hello";
    string str2 = "World";
    string result = str1 + " " + str2;  // Simple and safe
    
    // Dynamic sizing
    string dynamic;
    for (int i = 0; i < 1000; i++) 
    {
        dynamic += "a";  // Automatically resizes
    }
    
    // Easy comparison
    if (str1 == "Hello") // Direct comparison
    {  
        cout << "Match found!" << endl;
    }
    
    // Built-in functions
    cout << "Length: " << str1.length() << endl;
    cout << "Substring: " << result.substr(0, 5) << endl;
    
    return 0;
}
```

### Best Practices

- **Prefer std::string** over C-style strings in modern C++
- **Always initialize strings** to avoid garbage values
- **Check string bounds** to prevent buffer overflow
- **Use const char*** for string literals that won't change
- **Include proper headers**: `<string>` for std::string, `<cstring>` for C functions
