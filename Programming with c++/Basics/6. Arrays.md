
## 1. One-Dimensional Arrays

#### Basic Array Declaration

```cpp
#include <iostream>
using namespace std;

int main() 
{
    // Declaration and initialization
    int numbers[5] = {10, 20, 30, 40, 50};
    
    // Partial initialization (rest filled with 0)
    int scores[5] = {90, 85};  // {90, 85, 0, 0, 0}
    
    // Size determined by initializer
    int values[] = {1, 2, 3, 4};  // Size is 4
    
    // Declaration without initialization
    double prices[3];  // Contains garbage values
    
    // Accessing elements (0-indexed)
    cout << "First number: " << numbers[0] << endl;    // 10
    cout << "Last number: " << numbers[4] << endl;     // 50
    
    // Modifying elements
    numbers[2] = 35;
    cout << "Modified: " << numbers[2] << endl;        // 35
    
    return 0;
}
```

#### Array Traversal

```cpp
int main() 
{
    int grades[6] = {85, 92, 78, 94, 88, 76};
    int size = 6;
    
    // Forward traversal
    cout << "All grades: ";
    for (int i = 0; i < size; i++) 
    {
        cout << grades[i] << " ";
    }
    cout << endl;
    
    // Range-based for loop (C++11)
    cout << "Using range-based loop: ";
    for (int grade : grades) 
    {
        cout << grade << " ";
    }
    cout << endl;
    
    // Calculate average
    int sum = 0;
    for (int i = 0; i < size; i++) 
    {
        sum += grades[i];
    }
    
    double average = static_cast<double>(sum) / size; // Typecasting
    cout << "Average: " << average << endl;
    
    return 0;
}
```

####  Interchange the row and column of a matrix (Transpose)

```cpp
#include <iostream>
using namespace std;

int main() {
    int r, c;
    cout << "Enter rows and columns: ";
    cin >> r >> c;

    int mat[100][100];

    cout << "Enter matrix elements:\n";
    for (int i = 0; i < r; i++)
    {
        for (int j = 0; j < c; j++)
        {
            cin >> mat[i][j];
        }
    }

    cout << "Transpose Matrix:\n";
    for (int i = 0; i < c; i++) 
    {
        for (int j = 0; j < r; j++) 
        {
            cout << mat[j][i] << " "; // mat[j][i] keep in mind!
        }
        cout << endl;
    }
    return 0;
}
```

#### Add Two Matrices

```cpp
#include <iostream>
using namespace std;

int main() {
    int r, c;
    cout << "Enter rows and columns: ";
    cin >> r >> c;

    int A[100][100], B[100][100], Sum[100][100];

    cout << "Enter elements of Matrix A:\n";
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            cin >> A[i][j];

    cout << "Enter elements of Matrix B:\n";
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            cin >> B[i][j];

    // Add
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            Sum[i][j] = A[i][j] + B[i][j];

    cout << "Sum of matrices:\n";
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < c; j++)
            cout << Sum[i][j] << " ";
        cout << endl;
    }
    return 0;
}
```

#### Row sum and Column sum

```cpp
#include <iostream>
using namespace std;

int main() {
    int r, c;
    cout << "Enter rows and columns: ";
    cin >> r >> c;

    int mat[100][100];
    cout << "Enter matrix elements:\n";
    for (int i = 0; i < r; i++)
        for (int j = 0; j < c; j++)
            cin >> mat[i][j];

    cout << "Row sums:\n";
    for (int i = 0; i < r; i++) {
        int sum = 0;
        for (int j = 0; j < c; j++)
            sum += mat[i][j];
        cout << "Row " << i+1 << " = " << sum << endl;
    }

    cout << "Column sums:\n";
    for (int j = 0; j < c; j++) {
        int sum = 0;
        for (int i = 0; i < r; i++)
            sum += mat[i][j];
        cout << "Column " << j+1 << " = " << sum << endl;
    }
    return 0;
}
```

### 3. Array Initialization and Traversal

#### Different Initialization Methods

```cpp
int main() 
{
    // Method 1: Initialize at declaration
    int arr1[5] = {1, 2, 3, 4, 5};
    
    // Method 2: Initialize all to zero
    int arr2[5] = {0};  // All elements = 0
    int arr3[5] = {};   // Same as above
    
    // Method 3: Initialize after declaration
    int arr4[3];
    arr4[0] = 10;
    arr4[1] = 20;
    arr4[2] = 30;
    
    // Method 4: Using loops
    int arr5[5];
    for (int i = 0; i < 5; i++) 
    {
        arr5[i] = i * 10;  // 0, 10, 20, 30, 40
    }
    
    // Method 5: User input
    int size = 3;
    int userArray[size];
    cout << "Enter " << size << " numbers: ";
    for (int i = 0; i < size; i++) 
    {
        cin >> userArray[i];
    }
    
    return 0;
}
```

#### Advanced Traversal Techniques

```cpp
int main() {
    int numbers[] = {15, 8, 23, 4, 16, 42, 7, 12};
    int size = sizeof(numbers) / sizeof(numbers[0]);  // Calculate size
    
    // Find maximum
    int max = numbers[0];
    for (int i = 1; i < size; i++) 
    {
        if (numbers[i] > max) 
        {
            max = numbers[i];
        }
    }
    cout << "Maximum: " << max << endl;
    
    // Count even numbers
    int evenCount = 0;
    for (int i = 0; i < size; i++) 
    {
        if (numbers[i] % 2 == 0) 
        {
            evenCount++;
        }
    }
    cout << "Even numbers: " << evenCount << endl;
    
    // Reverse traversal
    cout << "Reversed: ";
    for (int i = size - 1; i >= 0; i--) 
    {
        cout << numbers[i] << " ";
    }
    cout << endl;
    
    return 0;
}
```
